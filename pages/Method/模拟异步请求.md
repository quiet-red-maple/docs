### 快速创建多数据数组方法

```js
const data =  Array(10).fill({
   name: '胡彦斌',
   age: 32,
   address: '西湖区湖底公园1号',
});
```

### 模拟异步请求

```js
const mockGetData = () =>  new Promise((reject) => {
    const data =  Array(10).fill(() => ({
      name: '胡彦斌',
      age: 32,
      address: '西湖区湖底公园1号',
    }));
    setTimeout(() => reject(data), 1000);
  }).then((res) => {
    return res;
  });


 const getList = async () => {
    const list = await mockGetData();
    console.log(50, list);
 };
```
